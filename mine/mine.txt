function mineLine()
	turtle.dig()
	turtle.forward()
	turtle.digUp()
	turtle.digDown()
end

function placeTorch()
	turtle.down()
	turtle.turnLeft()
	turtle.turnLeft()
	turtle.select(16)
	turtle.place()
	turtle.turnLeft()
	turtle.turnLeft()
	turtle.up()
end

function checkFuel()
	if turtle.getFuelLevel() < 80 then
		for i=0, 15 do
			turtle.select(i)
			if turtle.refuel() then
				break
			end
		end
	end
end

for i=0, 40 do
	if i%5==0 then
		checkFuel()
		placeTorch()
	end
	mineLine()
end

for i=0, 40 do
	if i%5==0 then
		checkFuel()
	end
	mineLine()
end

turtle.turnLeft()
turtle.forward()
turtle.turnLeft()
