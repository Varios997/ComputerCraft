function findChest()
	turtle.turnLeft()
	local success, data = turtle.inspectDown()
	local travel = 0
	while not ( (success) and (data.name == "minecraft:chest")) do
		turtle.forward()
		success, data = turtle.inspectDown()
		travel = travel + 1
	end
	
	for i=1, 15 do
		turtle.select(i)
		turtle.dropDown()
	end
	
	turtle.turnLeft()
	turtle.turnLeft()
	
	for i=1, travel do
		turtle.forward()
	end
	
	turtle.turnLeft()
end

function mineLine()
	turtle.dig()
	while not turtle.forward() do
		turtle.dig()
	end
	turtle.digUp()
	turtle.digDown()
end

function placeTorch()
	turtle.down()
	turtle.turnLeft()
	turtle.turnLeft()
	turtle.select(16)
	turtle.place()
	turtle.turnLeft()
	turtle.turnLeft()
	turtle.up()
end

function checkFuel()
	if turtle.getFuelLevel() < 80 then
		for i=1, 15 do
			turtle.select(i)
			if turtle.refuel() then
				break
			end
		end
	end
end

for i=0, 40 do
	if i%5==0 then
		checkFuel()
		placeTorch()
	end
	mineLine()
end

turtle.turnRight()
mineLine()
turtle.forward()
turtle.turnRight()

for i=0, 40 do
	if i%5==0 then
		checkFuel()
	end
	mineLine()
end

turtle.turnLeft()
turtle.forward()
turtle.turnLeft()

findChest()
